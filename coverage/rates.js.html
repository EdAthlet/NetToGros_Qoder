// Resolve a year key (number or string) into the rates object.
// Falls back to 2024 when an unknown year is supplied — same behaviour as the
// legacy file's `TAX_RATES[year] || TAX_RATES[2024]` pattern.
export function ratesForYear(year) {
  const key = typeof year === 'string' ? Number(year) : year;
  return TAX_RATES[key] || TAX_RATES[2024];
}
 
// Backwards-compatible side-effect version used by the legacy global API.
// Returns the resolved rates so callers can use the value directly instead of
// relying on module-level mutation.
export function updateTaxRatesForYear(year) {
  return ratesForYear(year);
}
