// Pure PAYE calculations (legacy js/calculator-core.js L151–178 and L396–492).
 
import { ratesForYear } from './rates.js';
import { roundToThree } from './rounding.js';
import { convertFromAnnual, normalisePeriodConfig } from './period.js';
import { calculateTaxCredits } from './credits.js';
import { formatCurrency } from './format.js';
 
function resolveStandardBand({ year, status, manualCutOff, PAYE_RATES }) {
  if (status === 'manual') {
    const yearKey = typeof year === 'string' ? year : String(year);
    const defaultCutOff = yearKey === '2024' ? 42000 : 44000;
    const parsed = manualCutOff != null ? parseFloat(manualCutOff) : NaN;
    return parsed || defaultCutOff;
  }
  if (status === 'married') return PAYE_RATES.standardBandMarried;
  if (status === 'marriedOneWorking') return PAYE_RATES.standardBandMarriedOneWorking;
  if (status === 'singleParent') return PAYE_RATES.standardBandSingleParent;
  return PAYE_RATES.standardBand;
}
 
/**
 * @param {number} grossIncome
 * @param {object} opts
 * @param {string|number} opts.year
 * @param {string} [opts.status='single']
 * @param {number|null} [opts.manualCutOff]
 * @returns {number}
 */
export function calculatePAYE(grossIncome, { year, status = 'single', manualCutOff } = {}) {
  const { PAYE_RATES } = ratesForYear(year);
  const standardBand = resolveStandardBand({ year, status, manualCutOff, PAYE_RATES });
 
  let paye;
  if (grossIncome <= standardBand) {
    paye = grossIncome * PAYE_RATES.standardRate;
  } else {
    paye = (standardBand * PAYE_RATES.standardRate) +
      ((grossIncome - standardBand) * PAYE_RATES.higherRate);
  }
  return roundToThree(Math.max(0, paye));
}
 
/**
 * @param {number} grossIncome
 * @param {object} opts
 * @param {string|number} opts.year
 * @param {string} [opts.status='single']
 * @param {number|null} [opts.manualCutOff]
 * @param {number|null} [opts.manualTaxCredits]
 * @param {object} opts.periodConfig
 */
export function calculatePAYEWithBreakdown(grossIncome, opts = {}) {
  const { year, status = 'single', manualCutOff, manualTaxCredits, periodConfig } = opts;
  const { PAYE_RATES } = ratesForYear(year);
  const period = normalisePeriodConfig(periodConfig);
 
  const standardBand = resolveStandardBand({ year, status, manualCutOff, PAYE_RATES });
 
  let taxCredits;
  if (status === 'manual') {
    const yearKey = typeof year === 'string' ? year : String(year);
    const defaultCredits = yearKey === '2024' ? 3750 : 4000;
    const parsed = manualTaxCredits != null ? parseFloat(manualTaxCredits) : NaN;
    taxCredits = parsed || defaultCredits;
  } else {
    taxCredits = calculateTaxCredits({ year, status });
  }
 
  const periodGross = convertFromAnnual(grossIncome, period);
  const periodStandardBand = convertFromAnnual(standardBand, period);
  const periodTaxCredits = convertFromAnnual(taxCredits, period);
 
  const payeBreakdown = {
    grossIncome: roundToThree(grossIncome),
    periodGross: roundToThree(periodGross),
    period: period.label,
    bands: [],
    grossTax: 0,
    taxCredits: roundToThree(taxCredits),
    periodTaxCredits: roundToThree(periodTaxCredits),
    standardBand: roundToThree(standardBand),
    periodStandardBand: roundToThree(periodStandardBand),
    netTax: 0,
    status
  };
 
  let totalGrossTax = 0;
 
  if (grossIncome > 0) {
    const standardRateIncome = Math.min(grossIncome, standardBand);
    const standardRateTax = standardRateIncome * PAYE_RATES.standardRate;
    const periodStandardRateIncome = convertFromAnnual(standardRateIncome, period);
    const periodStandardRateTax = convertFromAnnual(standardRateTax, period);
 
    payeBreakdown.bands.push({
      rate: PAYE_RATES.standardRate,
      rateDisplay: (PAYE_RATES.standardRate * 100).toFixed(0),
      range: `${formatCurrency(0)} \u2212 ${formatCurrency(periodStandardBand)}`,
      annualRange: `${formatCurrency(0)} \u2212 ${formatCurrency(standardBand)}`,
      taxableAmount: roundToThree(periodStandardRateIncome),
      annualTaxableAmount: roundToThree(standardRateIncome),
      tax: roundToThree(periodStandardRateTax),
      annualTax: roundToThree(standardRateTax),
      description: 'Standard rate'
    });
 
    totalGrossTax += standardRateTax;
  }
 
  const higherRateIncome = grossIncome - standardBand;
  const higherRateTax = higherRateIncome * PAYE_RATES.higherRate;
  const periodHigherRateIncome = convertFromAnnual(higherRateIncome, period);
  const periodHigherRateTax = convertFromAnnual(higherRateTax, period);
 
  if (higherRateIncome > 0) {
    payeBreakdown.bands.push({
      rate: PAYE_RATES.higherRate,
      rateDisplay: (PAYE_RATES.higherRate * 100).toFixed(0),
      range: `Over ${formatCurrency(periodStandardBand)}`,
      annualRange: `Over ${formatCurrency(standardBand)}`,
      taxableAmount: roundToThree(periodHigherRateIncome),
      annualTaxableAmount: roundToThree(higherRateIncome),
      tax: roundToThree(periodHigherRateTax),
      annualTax: roundToThree(higherRateTax),
      description: 'Higher rate'
    });
    totalGrossTax += higherRateTax;
  }
 
  payeBreakdown.grossTax = roundToThree(totalGrossTax);
  payeBreakdown.netTax = roundToThree(Math.max(0, totalGrossTax - taxCredits));
 
  return payeBreakdown;
}
